% Define a robust author name format that:
% - prints surname, given (Wang, Xiaochen) for all authors (consistent "surname, given")
% - bolds only when family==Wang AND given matches Xiaochen or initial X
\DeclareNameFormat{author}{%
  \nameparts{#1}% split name into parts
  \edef\mygiven{\namepartgiven}%
  \edef\myfamily{\namepartfamily}%
  % canonical target name pieces (edit here if you want different match)
  \def\targetFamily{Wang}%
  \def\targetGiven{Xiaochen}%
  \def\targetInit{X}% matches "Wang, X." or "Wang, X"
  %
  % compare family and given (allow exact given or single initial)
  \ifboolexpr{
    test {\ifstrequal{\myfamily}{\targetFamily}}
    and
    ( test {\ifstrequal{\mygiven}{\targetGiven}} or test {\ifstrequal{\mygiven}{\targetInit}} )
  }
    {% match -> bold surname and given
      \textbf{\namepartfamily}, \textbf{\namepartgiven}%
    }
    {% otherwise -> normal surname, given
      \namepartfamily, \namepartgiven%
    }%
  % comma between multiple authors
  \ifthenelse{\value{listcount}<\value{liststop}}{,\space}{}%
}
